-- dictionary tests

dict empty?
=> [true]

dict !$dict
$dict empty?
=> [true]
$dict "United States" 1776 put!
$dict length
=> [1]
$dict "United States" get
=> [1776]
$dict "Netherlands" 1648 put!
$dict length
=> [2]
$dict "Netherlands" get
=> [1648]
-- update an existing value:
$dict "Netherlands" 1815 put!
$dict length
=> [2]
$dict "Netherlands" get
=> [1815]
