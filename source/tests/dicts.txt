-- dictionary tests

dict empty?
=> [true]

dict !$dict
$dict empty?
=> [true]
$dict "United States" 1776 put!
$dict length
=> [1]
$dict "United States" get
=> [1776]
$dict "Netherlands" 1648 put!
$dict length
=> [2]
$dict "Netherlands" get
=> [1648]
-- update an existing value:
$dict "Netherlands" 1815 put!
$dict length
=> [2]
$dict "Netherlands" get
=> [1815]
$dict "Netherlands" lookup
=> [1815]
$dict "Netherlands" contains?
=> [true]
$dict "Germany" contains?
=> [false]

-- clear!
$dict clear!
$dict length
=> [0]

dict !$dict
$dict "a" 1 put!
$dict "b" 2 put!
$dict "c" 3 put!
$dict length
=> [3]
$dict "c" del!
$dict length
=> [2]
$dict/"b"  -- macro expansion: getter
=> [2]
4 $dict/!"d"  -- macro expansion: setter
$dict/"d"
=> [4]
