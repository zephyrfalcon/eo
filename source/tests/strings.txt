-- miscellaneous string tests

-- test str:upper
"abc" str:upper
=> ["ABC"]

"abc" ->string
=> ["abc"]
3 ->string
=> ["3"]

"abc" repr
=> ["\"abc\""]
-- includes the double quotes

-- escaping control characters
"abc\n"
=> ["abc\n"]
"abc\\n"
=> ["abc\\n"]
-- test if \x escaping is allowed
"a\x40b"
=> ["a@b"]
