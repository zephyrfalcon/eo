-- comparison of Eo objects

"abc" 1 cmp
=> [1]

-- integers
5 2 cmp
=> [1]
2 5 cmp
=> [-1]
9 9 cmp
=> [0]

-- strings
"hello" "bah" cmp
=> [1]
"hello" "zyxxy" cmp
=> [-1]
"hello" "hello" cmp
=> [0]
"hello" "hell" cmp
=> [1]

-- lists
[ 4 5 ] [ 1 2 ] cmp
=> [1]
[ 4 5 ] [ 1 ] cmp
=> [1]
[ 4 5 ] [ 6 ] cmp
=> [-1]
[ 3 3 ] [ 3 3 ] cmp
=> [0]
[ 3 3 ] [ 3 3 3 ] cmp
=> [-1]
[ "a" ] [ 1 ] cmp
=> [1]
