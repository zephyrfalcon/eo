-- miscellanous list tests

-- indexing
[ "a" "b" "c" ] 0 index
=> ["a"]

-- length
[ 1 2 3 ] length
=> [3]
[ ] length
=> [0]

-- adding in-place
[ 1 2 3 ] !$x
$x 4 add!
$x
=> [[ 1 2 3 4 ]]

-- map
[ 1 2 3 ] { 10 + } map
=> [[ 11 12 13 ]]

-- clear
[ 1 2 3 ] !$mylist
$mylist clear!
$mylist
=> [[ ]]

-- put!
[ 1 2 3 4 ] !$n
$n 0 33 put!
$n
=> [[ 33 2 3 4 ]]

-- del!
$n 0 del!
$n
=> [[ 2 3 4 ]]

-- contains?
[ 1 2 3 4 ] 3 contains?
=> [true]
[ 1 2 3 4 ] 9 contains?
=> [false]
