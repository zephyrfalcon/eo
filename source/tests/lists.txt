-- miscellanous list tests

-- indexing
[ "a" "b" "c" ] 0 index
=> ["a"]

-- length
[ 1 2 3 ] length
=> [3]
[ ] length
=> [0]

-- adding in-place
[ 1 2 3 ] !$x
$x 4 add!
$x
=> [[ 1 2 3 4 ]]

-- map
[ 1 2 3 ] { 10 + } map
=> [[ 11 12 13 ]]

-- filter
[ 1 2 3 4 ] { 2 > } filter
=> [[ 3 4 ]]

-- clear
[ 1 2 3 ] !$mylist
$mylist clear!
$mylist
=> [[ ]]

-- put!
[ 1 2 3 4 ] !$n
$n 0 33 put!
$n
=> [[ 33 2 3 4 ]]

-- del!
$n 0 del!
$n
=> [[ 2 3 4 ]]

-- contains?
[ 1 2 3 4 ] 3 contains?
=> [true]
[ 1 2 3 4 ] 9 contains?
=> [false]

-- reverse
[ 1 2 3 ] reverse
=> [[ 3 2 1 ]]
[ 4 5 6 ] !$list
$list reverse
=> [[ 6 5 4 ]]
-- make sure $list didn't change:
$list
=> [[ 4 5 6 ]]

-- reverse! does change the list in-place:
[ 7 8 9 ] !$list
$list reverse!
$list
=> [[ 9 8 7 ]]
